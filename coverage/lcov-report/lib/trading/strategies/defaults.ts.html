/**
 * Merge strategy with custom config
 */
export function createCustomStrategy(
  baseStrategy: Strategy,
  customConfig: Partial<Strategy['config']>
): Strategy {
  return {
    ...baseStrategy,
    id: `${baseStrategy.id}-custom-${Date.now()}`,
    config: {
      ...baseStrategy.config,
      ...customConfig
    }
  }
